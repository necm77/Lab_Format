% COMMANDS

% \pic{pic}{caption}{label}{width}
\newcommand{\pic}[4]{\begin{figure}[H]
\centering
\includegraphics[width = #4cm]{#1}
\caption{#2}
\label{#3}
\end{figure}}

% \twopics{pic1}{pic2}{caption1}{caption2}
\newcommand{\twopics}[4]{\begin{figure}[H]
  \centering
  \begin{minipage}[b]{0.45\textwidth}
    \includegraphics[width=\textwidth]{#1}
    \caption{#3}
  \end{minipage}
  \hfill
  \begin{minipage}[b]{0.45\textwidth}
    \includegraphics[width=\textwidth]{#2}
    \caption{#4}
  \end{minipage}
\end{figure}}
